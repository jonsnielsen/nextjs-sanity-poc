import { getCommercialDriver } from 'src/data/commercial-driver/commercialDriver.api';
// A simple example for testing it manually from your browser.
// If this is located at pages/api/preview.js, then
// open /api/preview from your browser.
export default async function handler(req, res) {
  // Check the secret and next parameters
  // This secret should only be known to this API route and the CMS
  if (
    req.query.secret !== process.env.SANITY_PREVIEW_SECRET ||
    !req.query.slug
  ) {
    return res.status(401).json({ message: 'Invalid token' });
  }

  console.log(req.query.slug);
  // const post = await getPostBySlug(req.query.slug);
  const commercialDriver = await getCommercialDriver(req.query.slug, true);
  // If the slug doesn't exist prevent preview mode from being enabled
  if (!commercialDriver) {
    return res.status(401).json({ message: 'Invalid slug' });
  }

  res.setPreviewData({});

  // Redirect to the path from the fetched post
  // We don't redirect to req.query.slug as that might lead to open redirect vulnerabilities
  res.writeHead(307, {
    Location: `/commercial-drivers/${commercialDriver.slug.current}`,
  });
  res.end();
  // res.end('Preview mode enabled');
}
